function toCapitalCase(str) {
  return str
    .split(/[\s-]+/)
    .map(word => word.charAt(0).toUpperCase() + word.slice(1).toLowerCase())
    .join(' ');
}

skills.map(skill => {
  const name = toCapitalCase(skill.name)
  const requiredSkills = skill.requiredSkills.map(toCapitalCase)
  return {
    ...skill,
    name,
    requiredSkills
  }
})
